009  
010  (defn tope
011    "no parents"
012    [x]
013    (nafc #(fresh [u] (child u %)) x))
014  
015  (defn yap-treeo
016    "Nodes which are yapped or sons of yapped node. No parent are empeached. Top
017    node is also included."
018    [q]
019    (all
020     (vertex q)
021     (nafc empeachedo q)
022     (fresh [p]
023       (conde [(tope q)]
024              [(kino p q)])
025       (nafc empeachedo p))))
026  
027  (defn yuk-treeo
028    [q]
029    (vertex q)
030    (conde [(yuk q)]
031           [(fresh [p]
032              (kino p q)
033              (yuk p))]))
034  
035  (defn availableo
036    [q]
037    (all
038      (vertex q)
039      (nafc inconsistento q)
040      (nafc yuk q)
041      (conda [(yap q)]
042             [(fresh [a b]
043                (kino a q)
044                (kino a b)
045                (kino b q)
046                (yuk a)
047                (yap b)
048                (l/!= a q)
049                (l/!= a b)
050                (l/!= b q))]
051             [(fresh [a]
052                (kino a q)
053                (empeachedo a)
054                (nafc yuk a)) (fresh [a b]
055                                (kino a q)
056                                (kino a b)
057                                (empeachedo a)
058                                (kino b q)
059                                (nafc yuk a)
060                                (yap b))]
061             [(nafc empeachedo q) (nafc yuk-treeo q)
062              (single-matcho #(fresh [a]
063                                (kino a q)
064                                (yap a)))]
065             [(empeachedo q) fail]
066             [(nafc yuk-treeo q)])))
