009  (defn- order-relationo
010    "Abstract the general pattern of a strict order relation in a logic,
011    relational way."
012    [relation x y]
013    (conde
014     [(relation x y)]
015     [(fresh [z]
016        (relation x z)
017        (order-relationo relation z y))]))

019  (def kino
020    "A goal where the two inputs x and y share kinship: x is an ancestor of y and
021    y a descandant of x."
022    (partial order-relationo child))
023  
024  (defn siblingso
025    [x s]
026    (fresh [z]
027      (child z x)
028      (child z s)
029      (l/!= x s)))

032    [q]
033    (all
034     (fresh [z]
035       (vertex z)
